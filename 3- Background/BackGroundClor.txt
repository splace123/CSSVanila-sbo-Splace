//TOPIC: Background Color CSS --

//HEADING: Background Color Property  --

//Use to set the background color of an element
//HEADING: Example --

body {
  background-color: red;
}
//TOPIC: There are 4 ways to set the  colors  --
//HEADING: Color Name  --

background-color: red;


//HEADING: Hexa Color --

background-color: #ff0000;

//HEADING: RGB --

background-color: rgb(255, 0, 0);

//HEADING: RGBA --

background-color: rgba(255, 0, 0, 0.5);

//IMPORTANT: A = Alpha (Opacity) --

Range: 0 to 1

0 = transparent
1 = fully visible
